// GC_CALENDAR_API_ROUTES_V1
type GcCalendarType = 'combo' | 'race' | 'training' | 'vote' | 'notice';
type GcCalendarEvent = {
  id: string;
  type: GcCalendarType;
  title: string;
  startsAt: string;
  endsAt: string | null;
  combo: string | null;
  track: string | null;
  cars: string[];
  link: string | null;
  description: string | null;
  enabled: boolean;
  featured: boolean;
  createdAt: string;
  updatedAt: string;
};

type GcCalendarStore = {
  version: 1;
  updatedAt: string;
  events: GcCalendarEvent[];
};

const gcCalendarTypes = new Set(['combo', 'race', 'training', 'vote', 'notice']);

function gcCalendarDataRoot() {
  const configured = process.env.APP_DATA_DIR?.trim();
  if (configured) return path.isAbsolute(configured) ? configured : path.join(rootDir, configured);
  return path.join(rootDir, 'data');
}

function getGcCalendarPath() {
  const configured = process.env.APP_CALENDAR_PATH?.trim();
  if (configured) return path.isAbsolute(configured) ? configured : path.join(rootDir, configured);
  return path.join(gcCalendarDataRoot(), 'app/calendar-events.json');
}

function createEmptyGcCalendarStore(): GcCalendarStore {
  return { version: 1, updatedAt: new Date().toISOString(), events: [] };
}

function readGcCalendarStore(): GcCalendarStore {
  const filePath = getGcCalendarPath();
  if (!fs.existsSync(filePath)) return createEmptyGcCalendarStore();
  try {
    const parsed = JSON.parse(fs.readFileSync(filePath, 'utf8')) as Partial<GcCalendarStore>;
    return {
      version: 1,
      updatedAt: String(parsed.updatedAt || new Date().toISOString()),
      events: Array.isArray(parsed.events) ? parsed.events.map(sanitizeGcCalendarEvent).filter(Boolean) as GcCalendarEvent[] : []
    };
  } catch (error) {
    console.error('[GC] Error leyendo calendar-events.json:', error);
    return createEmptyGcCalendarStore();
  }
}

function writeGcCalendarStore(store: GcCalendarStore) {
  const filePath = getGcCalendarPath();
  fs.mkdirSync(path.dirname(filePath), { recursive: true });
  const nextStore = { version: 1 as const, updatedAt: new Date().toISOString(), events: store.events };
  const tempPath = `${filePath}.tmp`;
  fs.writeFileSync(tempPath, `${JSON.stringify(nextStore, null, 2)}\n`, 'utf8');
  fs.renameSync(tempPath, filePath);
}

function gcCalendarText(value: unknown, fallback = '') {
  const text = String(value ?? '').trim();
  return text || fallback;
}

function gcCalendarIso(value: unknown) {
  const text = gcCalendarText(value);
  if (!text) return '';
  const parsed = new Date(text);
  return Number.isNaN(parsed.getTime()) ? '' : parsed.toISOString();
}

function sanitizeGcCalendarEvent(input: any): GcCalendarEvent | null {
  const now = new Date().toISOString();
  const type = gcCalendarTypes.has(String(input?.type)) ? String(input.type) as GcCalendarType : 'notice';
  const title = gcCalendarText(input?.title).slice(0, 160);
  const startsAt = gcCalendarIso(input?.startsAt);
  if (!title || !startsAt) return null;
  const cars = Array.isArray(input?.cars)
    ? input.cars.map((item: unknown) => gcCalendarText(item)).filter(Boolean).slice(0, 24)
    : gcCalendarText(input?.cars).split(',').map((item) => item.trim()).filter(Boolean).slice(0, 24);
  return {
    id: gcCalendarText(input?.id) || crypto.randomUUID(),
    type,
    title,
    startsAt,
    endsAt: gcCalendarIso(input?.endsAt) || null,
    combo: gcCalendarText(input?.combo).slice(0, 180) || null,
    track: gcCalendarText(input?.track).slice(0, 120) || null,
    cars,
    link: gcCalendarText(input?.link).slice(0, 500) || null,
    description: gcCalendarText(input?.description).slice(0, 1200) || null,
    enabled: input?.enabled !== false,
    featured: Boolean(input?.featured),
    createdAt: gcCalendarIso(input?.createdAt) || now,
    updatedAt: now
  };
}

async function gcCalendarIsAdminRequest(req: any) {
  const cookie = String(req.headers?.cookie || '');
  if (!cookie) return false;
  const base = `http://127.0.0.1:${PORT}`;
  for (const endpoint of ['/api/auth/status', '/api/profile', '/api/admin/status']) {
    try {
      const response = await fetch(`${base}${endpoint}`, { headers: { cookie } });
      if (!response.ok) continue;
      const data: any = await response.json();
      const role = data?.user?.role || data?.currentUser?.role || data?.profile?.role || data?.role;
      if (role === 'admin' || data?.admin === true || data?.authorized === true) return true;
    } catch {
      // seguimos probando otros endpoints
    }
  }
  return false;
}

async function gcCalendarRequireAdmin(req: any, res: any, next: any) {
  const allowed = await gcCalendarIsAdminRequest(req);
  if (!allowed) return res.status(403).json({ ok: false, message: 'Solo administradores.' });
  return next();
}

function sortGcCalendarEvents(events: GcCalendarEvent[]) {
  return [...events].sort((a, b) => String(a.startsAt).localeCompare(String(b.startsAt)) || String(a.title).localeCompare(String(b.title), 'es'));
}

app.get('/api/calendar/events', (_req, res) => {
  const store = readGcCalendarStore();
  res.json({ ok: true, updatedAt: store.updatedAt, events: sortGcCalendarEvents(store.events.filter((event) => event.enabled !== false)) });
});

app.get('/api/admin/calendar/events', gcCalendarRequireAdmin, (_req, res) => {
  const store = readGcCalendarStore();
  res.json({ ok: true, storage: { path: getGcCalendarPath() }, updatedAt: store.updatedAt, events: sortGcCalendarEvents(store.events) });
});

app.post('/api/admin/calendar/events', gcCalendarRequireAdmin, (req, res) => {
  const store = readGcCalendarStore();
  const nextEvent = sanitizeGcCalendarEvent(req.body || {});
  if (!nextEvent) return res.status(400).json({ ok: false, message: 'Título e inicio son obligatorios.' });
  const index = store.events.findIndex((event) => event.id === nextEvent.id);
  if (index >= 0) {
    nextEvent.createdAt = store.events[index].createdAt || nextEvent.createdAt;
    store.events[index] = nextEvent;
  } else {
    store.events.push(nextEvent);
  }
  writeGcCalendarStore(store);
  res.json({ ok: true, event: nextEvent, events: sortGcCalendarEvents(store.events) });
});

app.delete('/api/admin/calendar/events/:id', gcCalendarRequireAdmin, (req, res) => {
  const store = readGcCalendarStore();
  const id = String(req.params.id || '');
  const before = store.events.length;
  store.events = store.events.filter((event) => event.id !== id);
  writeGcCalendarStore(store);
  res.json({ ok: true, deleted: before !== store.events.length, events: sortGcCalendarEvents(store.events) });
});
